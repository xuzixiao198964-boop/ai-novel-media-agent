#### 1.2.2 小说生成 Pipeline

```python
class NovelPipeline:
    """增强版小说生成流水线"""

    STANDARD_AGENTS = [
        "trend", "style", "planner", "writer", "polish", "auditor", "reviser"
    ]
    ENHANCED_AGENTS = [
        "conflict",       # 冲突强度校验（微/短/中篇）
        "uniqueness",     # 策划唯一性校验
        "video_review",   # 视频适配评审
    ]

    async def run(self, task: Task):
        # 阶段一：分析与策划
        trend_result = await self.run_agent("trend", task)
        style_result = await self.run_agent("style", task)

        # 策划时注入大咖建议
        task.context["advice_refs"] = await self.advice_matcher.match(task.config)
        planner_result = await self.run_agent("planner", task)

        # 唯一性校验
        await self.run_agent("uniqueness", task)

        # 阶段二：写作与润色
        await self.run_agent("writer", task)
        await self.run_agent("polish", task)

        # 阶段三：审计与修订
        audit_result = await self.run_agent("auditor", task)

        # 冲突强度校验（微/短/中篇必须）
        if task.config.length_type in ("micro", "short", "medium"):
            await self.run_agent("conflict", task)

        await self.run_agent("reviser", task)

        # 阶段四：视频适配评审（若需生成视频）
        if task.needs_video:
            await self.run_agent("video_review", task)

        # 阶段五：计费
        await self.billing.charge(task)
```

#### 1.2.3 大咖建议匹配器

```python
class AdviceMatcher:
    """从建议库中为每本小说匹配不冲突的建议组合"""

    async def match(self, config: NovelConfig) -> List[AdviceRef]:
        # 1. 根据题材筛选候选建议
        candidates = self.library.query(genre=config.genre, sub_genre=config.sub_genre)

        # 2. 随机选择 1-3 条建议
        selected = random.sample(candidates, k=min(3, len(candidates)))

        # 3. 冲突检测：检查建议之间是否矛盾
        while self.has_conflict(selected):
            selected = self.resolve_conflict(selected, candidates)

        # 4. 记录引用（确保可追溯）
        return [AdviceRef(id=a.id, source=a.source, summary=a.summary) for a in selected]
```

#### 1.2.4 付费与计费

```python
class BillingEngine:
    """计费引擎：用户消耗 = API 成本 × 倍率"""

    MARKUP_RANGE = (1.1, 1.2)

    async def estimate_cost(self, task_config: TaskConfig) -> CostEstimate:
        """预估任务成本（创建任务前调用）"""
        novel_cost = self._estimate_novel(task_config.novel_config)
        video_cost = self._estimate_video(task_config.video_config)
        total_api = novel_cost + video_cost
        markup = random.uniform(*self.MARKUP_RANGE)
        return CostEstimate(
            api_cost=total_api,
            user_cost=round(total_api * markup, 2),
            breakdown={"novel": novel_cost, "video": video_cost}
        )

    async def charge(self, task: Task):
        """任务完成后按实际消耗扣费"""
        actual_api_cost = await self._calculate_actual_cost(task)
        markup = random.uniform(*self.MARKUP_RANGE)
        user_charge = round(actual_api_cost * markup, 2)

        await self.deduct_balance(task.user_id, user_charge)
        await self.record_consumption(task, actual_api_cost, user_charge)
```

```python
class QueueManager:
    """任务排队管理"""

    async def get_queue_info(self, task_id: str) -> QueueInfo:
        position = await self.redis.zrank("task_queue", task_id)
        if position is None:
            return QueueInfo(status="running", position=0, est_wait=0)

        avg_time = await self._get_avg_completion_time()
        return QueueInfo(
            status="queued",
            position=position + 1,
            est_wait=int(position * avg_time),
            est_wait_display=self._format_duration(position * avg_time)
        )
```
